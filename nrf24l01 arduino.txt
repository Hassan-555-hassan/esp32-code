#include <SPI.h>
#include <RF24.h>

#define BUTTON_PIN 2
#define CE_PIN 9
#define CSN_PIN 8

RF24 radio(CE_PIN, CSN_PIN);
const byte address[6] = "00001";

void setup() {
  Serial.begin(9600);
  pinMode(BUTTON_PIN, INPUT_PULLUP);

  Serial.println("‚è≥ Initializing Transmitter...");

  if (!radio.begin()) {
    Serial.println("‚ùå NRF24L01 not detected!");
    while (1);
  }

  radio.openWritingPipe(address);
  radio.setChannel(100);
  radio.setPALevel(RF24_PA_LOW);
  radio.setDataRate(RF24_1MBPS);
  radio.stopListening();

  Serial.println("‚úÖ Transmitter Ready");
}

void loop() {
  if (digitalRead(BUTTON_PIN) == LOW) {
    const char text[] = "LIGHT";
    bool success = radio.write(&text, sizeof(text));
    
    if (success) {
      Serial.println("üì§ Sent: LIGHT");
    } else {
      Serial.println("‚ùå Send Failed!");
    }
    
    delay(500);
  }
}




#include <SPI.h>
#include <RF24.h>

#define LED_PIN 10
#define CE_PIN 9
#define CSN_PIN 8

RF24 radio(CE_PIN, CSN_PIN);
const byte address[6] = "00001";

void setup() {
  Serial.begin(9600);
  pinMode(LED_PIN, OUTPUT);
  digitalWrite(LED_PIN, LOW);

  Serial.println("‚è≥ Initializing Receiver...");

  if (!radio.begin()) {
    Serial.println("‚ùå NRF24L01 not detected!");
    while (1);
  }

  radio.openReadingPipe(0, address);
  radio.setChannel(100);
  radio.setPALevel(RF24_PA_LOW);
  radio.setDataRate(RF24_1MBPS);
  radio.startListening();

  Serial.println("‚úÖ Receiver Ready. Listening...");
}

void loop() {
  if (radio.available()) {
    char text[32] = "";
    radio.read(&text, sizeof(text));

    Serial.print("üì• Received: ");
    Serial.println(text);

    if (strcmp(text, "LIGHT") == 0) {
      digitalWrite(LED_PIN, HIGH);
      delay(500);
      digitalWrite(LED_PIN, LOW);
    }
  }
}
